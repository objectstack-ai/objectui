## Schema

```plaintext
interface FlexSchema {
  type: 'flex';
  direction?: 'row' | 'column';
  gap?: number;
  align?: 'start' | 'center' | 'end';
  justify?: 'start' | 'center' | 'end' | 'between';
  wrap?: boolean;
  children: SchemaNode[];
  className?: string;
}
```
