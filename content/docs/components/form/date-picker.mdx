/>

## Schema

```plaintext
interface DatePickerSchema {
  type: 'date-picker';
  value?: string | Date;          // Selected date (ISO string or Date)
  placeholder?: string;           // Placeholder text
  
  // Events
  onDateChange?: string | ActionConfig;
  
  // States
  disabled?: boolean;
  
  // Styling
  className?: string;
}
```
