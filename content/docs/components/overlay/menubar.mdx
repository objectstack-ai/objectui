## Schema

```plaintext
interface MenubarItem {
  label?: string;
  value?: string;
  icon?: string;
  shortcut?: string[];
  type?: 'separator';
  disabled?: boolean;
}

interface MenubarMenu {
  label: string;
  items: MenubarItem[];
}

interface MenubarSchema {
  type: 'menubar';
  menus: MenubarMenu[];           // Menu definitions
  
  // Events
  onSelect?: string | ActionConfig;
  
  // Styling
  className?: string;
}
```
