The Context Menu component displays a menu when right-clicking on an element.

## Basic Usage

<ComponentDemo
  schema={{
    type: 'context-menu',
    trigger: {
      type: 'card',
      content: 'Right-click here',
      className: 'p-8 text-center border-dashed'
    },
    items: [
      { label: 'Copy', value: 'copy', icon: 'copy' },
      { label: 'Cut', value: 'cut', icon: 'scissors' },
      { label: 'Paste', value: 'paste', icon: 'clipboard' },
      { type: 'separator' },
      { label: 'Delete', value: 'delete', icon: 'trash' }
    ]
  }}
  title="Basic Context Menu"
/>

## Schema

```plaintext
interface ContextMenuItem {
  label?: string;
  value?: string;
  icon?: string;
  type?: 'separator';
  disabled?: boolean;
}

interface ContextMenuSchema {
  type: 'context-menu';
  trigger: ComponentSchema;       // Trigger element
  items: ContextMenuItem[];       // Menu items
  onSelect?: string | ActionConfig;
  className?: string;
}
```
