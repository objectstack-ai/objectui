The Dropdown Menu component displays a list of actions or options when triggered.

## Basic Usage

<ComponentDemo
  schema={{
    type: 'dropdown-menu',
    trigger: {
      type: 'button',
      label: 'Open Menu'
    },
    items: [
      { label: 'Profile', value: 'profile' },
      { label: 'Settings', value: 'settings' },
      { type: 'separator' },
      { label: 'Logout', value: 'logout' }
    ]
  }}
  title="Basic Dropdown Menu"
/>

## Schema

```plaintext
interface DropdownMenuItem {
  label?: string;
  value?: string;
  icon?: string;
  variant?: 'default' | 'destructive';
  type?: 'separator';
  disabled?: boolean;
}

interface DropdownMenuSchema {
  type: 'dropdown-menu';
  trigger: ComponentSchema;       // Trigger component
  items: DropdownMenuItem[];      // Menu items
  
  // Events
  onSelect?: string | ActionConfig;
  
  // Styling
  className?: string;
}
```
